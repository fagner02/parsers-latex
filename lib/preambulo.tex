\newcommand{\removezero}[1]{%
\StrMid{#1}{1}{1}[\zero]%
\StrMid{#1}{2}{3}[\inte]%
\ifthenelse{\equal{0}{\zero}}{\inte}{#1}%
}
\newcommand{\usedate}[1]{%
\def\x{#1}%
\setsepchar[.]{-}%
\readlist\datee{\x}%
{\removezero{\datee[2]} \monthname{\datee[3]} \datee[1]}%
}
