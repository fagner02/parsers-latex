\begin{apendice}{APÊNDICE B -- ALGORITMO FOLLOW}
	\label{ap:follow}

	\begin{algorithm}[htbp]
		\caption{Follow}
		\Input{Gramática $G$, Símbolo $A$}
		\Output{Conjunto $\mathit{follow}$}
		\Inicio{
			$\mathit{follow} \gets \varnothing$ \\

			\Ifx{$A$ é o símbolo inicial $S$}{
				$\mathit{follow} \gets \{ \$ \}$
			}

			\Repeat{não haja mais mudanças}{
				\ForEach{produção $B \rightarrow \alpha A \beta$ em $G$}{
					\Ifx{$\beta \neq \epsilon$}{
						$\mathit{follow} \gets \mathit{follow} \cup \left( \textsc{First}(\beta) - \{ \epsilon \} \right)$
					}
					\Ifx{$\beta = \epsilon \lor \epsilon \in \textsc{First}(\beta)$}{
						$\mathit{follow} \gets \mathit{follow} \cup \textsc{Follow}(B)$
					}
				}
			}
			\Return{$\mathit{follow}$}
		}
	\end{algorithm}
\end{apendice}
